Joignez toutes les chaînes de caractères données, puis remplacez "right" par "left" et renvoyez (n'imprimez pas) le résultat. 

def left_join(phrases: tuple[str]) -> str:

    join_phrase = ','.join(phrases)
    result = join_phrase.replace('right','left')

    return result
